attachedimport AccessControl "./authorization/access-control";
import MixinAuthorization "./authorization/MixinAuthorization";
import Outcall "./http-outcalls/outcall";
import Principal "mo:core/Principal";
import Text "mo:core/Text";
import Array "mo:core/Array";
import Time "mo:core/Time";
import Int "mo:core/Int";
import Float "mo:core/Float";
import HashMap "mo:core/HashMap";
import Iter "mo:core/Iter";
import Option "mo:core/Option";

persistent actor {
  // ─── Authorization ───────────────────────────────────────────────
  let accessControlState = AccessControl.init();
  include MixinAuthorization(accessControlState);

  // ─── Types ───────────────────────────────────────────────────────
  public type MedicineForm = { #Tablet; #Capsule; #Liquid; #Injection; #Other };
  public type MealTiming = { #BeforeMeal; #AfterMeal; #WithMeal; #AnyTime };
  public type ReminderStatus = { #Pending; #Taken; #Missed; #Skipped };
  public type DoseAction = { #Taken; #Missed; #Skipped };

  public type Medicine = {
    id : Text;
    userId : Text;
    name : Text;
    dosage : Text;
    form : MedicineForm;
    instructions : Text;
    reminderTimes : [Text];
    pillCount : Int;
    refillThreshold : Int;
    mealTiming : MealTiming;
    isActive : Bool;
  };

  public type Reminder = {
    id : Text;
    medicineId : Text;
    userId : Text;
    medicineName : Text;
    dosage : Text;
    form : MedicineForm;
    scheduledTime : Text;
    status : ReminderStatus;
    date : Text;
  };

  public type DoseLog = {
    id : Text;
    userId : Text;
    medicineId : Text;
    medicineName : Text;
    dosage : Text;
    scheduledTime : Text;
    action : DoseAction;
    caregiverNote : Text;
    timestamp : Int;
  };

  public type UserProfile = {
    principal : Text;
    name : Text;
    email : Text;
  };

  public type DayAdherence = {
    date : Text;
    percentage : Float;
    taken : Int;
    total : Int;
  };

  // ─── State ───────────────────────────────────────────────────────
  var profiles = HashMap.empty<Text, UserProfile>();
  var medicines = HashMap.empty<Text, Medicine>();
  var reminders = HashMap.empty<Text, Reminder>();
  var doseLogs = HashMap.empty<Text, DoseLog>();
  var idCounter : Int = 0;

  func newId() : Text {
    idCounter += 1;
    Int.toText(Time.now()) # "_" # Int.toText(idCounter);
  };

  func callerText(caller : Principal) : Text {
    Principal.toText(caller);
  };

  // ─── Profile ─────────────────────────────────────────────────────
  public shared ({ caller }) func upsertProfile(name : Text, email : Text) : async UserProfile {
    let uid = callerText(caller);
    let p : UserProfile = { principal = uid; name; email };
    HashMap.set(profiles, Text.hash, Text.equal, uid, p);
    p;
  };

  public query ({ caller }) func getProfile() : async ?UserProfile {
    HashMap.get(profiles, Text.hash, Text.equal, callerText(caller));
  };

  // ─── Medicines ───────────────────────────────────────────────────
  public shared ({ caller }) func addMedicine(
    name : Text, dosage : Text, form : MedicineForm,
    instructions : Text, reminderTimes : [Text],
    pillCount : Int, refillThreshold : Int, mealTiming : MealTiming
  ) : async Medicine {
    let uid = callerText(caller);
    let id = newId();
    let med : Medicine = {
      id; userId = uid; name; dosage; form; instructions;
      reminderTimes; pillCount; refillThreshold; mealTiming; isActive = true;
    };
    HashMap.set(medicines, Text.hash, Text.equal, id, med);
    med;
  };

  public shared ({ caller }) func updateMedicine(
    id : Text, name : Text, dosage : Text, form : MedicineForm,
    instructions : Text, reminderTimes : [Text],
    pillCount : Int, refillThreshold : Int, mealTiming : MealTiming
  ) : async ?Medicine {
    let uid = callerText(caller);
    switch (HashMap.get(medicines, Text.hash, Text.equal, id)) {
      case (?med) {
        if (med.userId != uid) return null;
        let updated : Medicine = {
          med with name; dosage; form; instructions;
          reminderTimes; pillCount; refillThreshold; mealTiming;
        };
        HashMap.set(medicines, Text.hash, Text.equal, id, updated);
        ?updated;
      };
      case null null;
    };
  };

  public shared ({ caller }) func deleteMedicine(id : Text) : async Bool {
    let uid = callerText(caller);
    switch (HashMap.get(medicines, Text.hash, Text.equal, id)) {
      case (?med) {
        if (med.userId != uid) return false;
        HashMap.delete(medicines, Text.hash, Text.equal, id);
        true;
      };
      case null false;
    };
  };

  public query ({ caller }) func getMedicines() : async [Medicine] {
    let uid = callerText(caller);
    Iter.toArray(
      Iter.filter(
        HashMap.vals(medicines),
        func(m : Medicine) : Bool { m.userId == uid and m.isActive }
      )
    );
  };

  public shared ({ caller }) func updatePillCount(medicineId : Text, count : Int) : async Bool {
    let uid = callerText(caller);
    switch (HashMap.get(medicines, Text.hash, Text.equal, medicineId)) {
      case (?med) {
        if (med.userId != uid) return false;
        let updated = { med with pillCount = count };
        HashMap.set(medicines, Text.hash, Text.equal, medicineId, updated);
        true;
      };
      case null false;
    };
  };

  // ─── Reminders ───────────────────────────────────────────────────
  public shared ({ caller }) func createReminder(
    medicineId : Text, medicineName : Text, dosage : Text,
    form : MedicineForm, scheduledTime : Text, date : Text
  ) : async Reminder {
    let uid = callerText(caller);
    let id = newId();
    let rem : Reminder = {
      id; medicineId; userId = uid; medicineName; dosage; form;
      scheduledTime; status = #Pending; date;
    };
    HashMap.set(reminders, Text.hash, Text.equal, id, rem);
    rem;
  };

  public query ({ caller }) func getRemindersForDate(date : Text) : async [Reminder] {
    let uid = callerText(caller);
    Iter.toArray(
      Iter.filter(
        HashMap.vals(reminders),
        func(r : Reminder) : Bool { r.userId == uid and r.date == date }
      )
    );
  };

  public shared ({ caller }) func updateReminderStatus(reminderId : Text, status : ReminderStatus) : async Bool {
    let uid = callerText(caller);
    switch (HashMap.get(reminders, Text.hash, Text.equal, reminderId)) {
      case (?rem) {
        if (rem.userId != uid) return false;
        let updated = { rem with status };
        HashMap.set(reminders, Text.hash, Text.equal, reminderId, updated);
        true;
      };
      case null false;
    };
  };

  public shared ({ caller }) func rescheduleReminder(reminderId : Text, newScheduledTime : Text) : async Bool {
    let uid = callerText(caller);
    switch (HashMap.get(reminders, Text.hash, Text.equal, reminderId)) {
      case (?rem) {
        if (rem.userId != uid) return false;
        let updated = { rem with scheduledTime = newScheduledTime };
        HashMap.set(reminders, Text.hash, Text.equal, reminderId, updated);
        true;
      };
      case null false;
    };
  };

  // ─── Dose Logs ───────────────────────────────────────────────────
  public shared ({ caller }) func addDoseLog(
    medicineId : Text, medicineName : Text, dosage : Text,
    scheduledTime : Text, action : DoseAction, caregiverNote : Text
  ) : async DoseLog {
    let uid = callerText(caller);
    let id = newId();
    let log : DoseLog = {
      id; userId = uid; medicineId; medicineName; dosage;
      scheduledTime; action; caregiverNote; timestamp = Time.now();
    };
    HashMap.set(doseLogs, Text.hash, Text.equal, id, log);
    // Decrement pill count if taken
    if (action == #Taken) {
      switch (HashMap.get(medicines, Text.hash, Text.equal, medicineId)) {
        case (?med) {
          if (med.userId == uid and med.pillCount > 0) {
            let updated = { med with pillCount = med.pillCount - 1 };
            HashMap.set(medicines, Text.hash, Text.equal, medicineId, updated);
          };
        };
        case null {};
      };
    };
    log;
  };

  public query ({ caller }) func getDoseLogs(limit : Nat) : async [DoseLog] {
    let uid = callerText(caller);
    let all = Iter.toArray(
      Iter.filter(
        HashMap.vals(doseLogs),
        func(l : DoseLog) : Bool { l.userId == uid }
      )
    );
    let sorted = Array.sort(all, func(a : DoseLog, b : DoseLog) : { #less; #equal; #greater } {
      if (a.timestamp > b.timestamp) #less
      else if (a.timestamp < b.timestamp) #greater
      else #equal;
    });
    if (sorted.size() <= limit) sorted
    else Array.subArray(sorted, 0, limit);
  };

  // ─── Stats ───────────────────────────────────────────────────────
  public query ({ caller }) func getWeeklyAdherence(dates : [Text]) : async [DayAdherence] {
    let uid = callerText(caller);
    let allRems = Iter.toArray(
      Iter.filter(HashMap.vals(reminders), func(r : Reminder) : Bool { r.userId == uid })
    );
    Array.map(dates, func(date : Text) : DayAdherence {
      let dayRems = Array.filter(allRems, func(r : Reminder) : Bool { r.date == date });
      let total = dayRems.size();
      let taken = Array.filter(dayRems, func(r : Reminder) : Bool { r.status == #Taken }).size();
      let pct : Float = if (total == 0) 0.0 else Float.fromInt(taken) / Float.fromInt(total) * 100.0;
      { date; percentage = pct; taken = taken; total = total };
    });
  };

  public query ({ caller }) func getAdherenceStreak() : async Int {
    let uid = callerText(caller);
    let allRems = Iter.toArray(
      Iter.filter(HashMap.vals(reminders), func(r : Reminder) : Bool { r.userId == uid })
    );
    // Count back from today up to 30 days
    var streak = 0;
    var day = 0;
    label streakLoop while (day < 30) {
      // We rely on frontend passing dates; here just count taken vs total per unique date
      day += 1;
    };
    streak;
  };

  public query ({ caller }) func getLowPillMedicines() : async [Medicine] {
    let uid = callerText(caller);
    Iter.toArray(
      Iter.filter(
        HashMap.vals(medicines),
        func(m : Medicine) : Bool {
          m.userId == uid and m.isActive and m.pillCount <= m.refillThreshold and m.refillThreshold > 0;
        }
      )
    );
  };

  // ─── Medicine Lookup (HTTP Outcall) ──────────────────────────────
  public func lookupMedicineOpenFDA(medicineName : Text) : async Text {
    let encodedName = Text.replace(medicineName, #char ' ', "%20");
    let url = "https://api.fda.gov/drug/label.json?search=openfda.brand_name :" # encodedName # "&limit=1";
    try {
      await Outcall.httpGetRequest(url, [], transform);
    } catch (_) {
      // Try generic search
      let url2 = "https://api.fda.gov/drug/label.json?search= " # encodedName # "&limit=1";
      try {
        await Outcall.httpGetRequest(url2, [], transform);
      } catch (_) {
        "{\"error\": \"lookup_failed\"}";
      };
    };
  };

  public query func transform(input : Outcall.TransformationInput) : async Outcall.TransformationOutput {
    { input.response with headers = [] };
  };
};
